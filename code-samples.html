1. React Component - Portfolio Card

A reusable React component for displaying portfolio projects with props for title, description, and link.

import React from 'react';

const PortfolioCard = ({ title, description, link, image, tags }) => {
  return (
    <div className="portfolio-card">

          <img src={image} alt={title} />
        </div>
      )}
      <div className="card-content">
        <h3 className="card-title">{title}</h3>
        <p className="card-description">{description}</p>
        {tags && (
          <div className="card-tags">
            {tags.map((tag, index) => (
              <span key={index} className="tag">{tag}</span>
            ))}
          </div>
        )}
        <a href={link} target="_blank" rel="noopener noreferrer" className="card-link">

        </a>

  );

  tags: []

3. Python - Machine Learning Pipeline

A complete machine learning pipeline using scikit-learn for data preprocessing, model training, and evaluation with RandomForest classifier.

import pandas as pd
import numpy as np
from sklearn.model_selection import train_test_split, cross_val_score
from sklearn.ensemble import RandomForestClassifier
from sklearn.preprocessing import StandardScaler, LabelEncoder
from sklearn.metrics import accuracy_score, classification_report, confusion_matrix
import joblib

class MLPipeline:
    def __init__(self, data_path):
        """Initialize the ML pipeline with data path"""
        self.data_path = data_path
        self.model = None
        self.scaler = StandardScaler()
        self.label_encoders = {}
        
    def load_data(self):
        """Load and display basic info about the dataset"""
        self.df = pd.read_csv(self.data_path)
        print(f"Dataset Shape: {self.df.shape}")
        print(f"\nColumn Types:\n{self.df.dtypes}")
        print(f"\nMissing Values:\n{self.df.isnull().sum()}")
        return self.df
    
    def preprocess(self, target_column):
        """Preprocess the data"""
        # Handle missing values
        self.df = self.df.dropna()
        
        # Encode categorical variables
        categorical_cols = self.df.select_dtypes(include=['object']).columns
        
        for col in categorical_cols:
            if col != target_column:
                self.label_encoders[col] = LabelEncoder()
                self.df[col] = self.label_encoders[col].fit_transform(self.df[col])
        
        # Separate features and target
        X = self.df.drop(target_column, axis=1)
        y = self.df[target_column]
        
        # Scale features
        X_scaled = self.scaler.fit_transform(X)
        
        return X_scaled, y
    
    def train_model(self, X, y, test_size=0.2):
        """Train the model and return metrics"""
        # Split data
        X_train, X_test, y_train, y_test = train_test_split(
            X, y, test_size=test_size, random_state=42, stratify=y
        )
        
        # Initialize and train model
        self.model = RandomForestClassifier(
            n_estimators=100,
            max_depth=10,
            random_state=42,
            n_jobs=-1
        )
        
        self.model.fit(X_train, y_train)
        
        # Make predictions
        y_pred = self.model.predict(X_test)
        
        # Calculate metrics
        accuracy = accuracy_score(y_test, y_pred)
        cv_scores = cross_val_score(self.model, X, y, cv=5)
        
        print(f"\n{'='*50}")
        print("Model Performance Metrics")
        print(f"{'='*50}")
        print(f"Accuracy: {accuracy:.4f}")
        print(f"Cross-Validation Score: {cv_scores.mean():.4f} (+/- {cv_scores.std()*2:.4f})")
        print(f"\nClassification Report:\n{classification_report(y_test, y_pred)}")
        print(f"\nConfusion Matrix:\n{confusion_matrix(y_test, y_pred)}")
        
        return {
            'accuracy': accuracy,
            'cv_scores': cv_scores,
            'predictions': y_pred
        }
    
    def save_model(self, model_path, scaler_path):
        """Save trained model and scaler"""
        joblib.dump(self.model, model_path)
        joblib.dump(self.scaler, scaler_path)
        print(f"\nModel saved to {model_path}")
        print(f"Scaler saved to {scaler_path}")
    
    def load_model(self, model_path, scaler_path):
        """Load trained model and scaler"""
        self.model = joblib.load(model_path)
        self.scaler = joblib.load(scaler_path)
        print("Model and scaler loaded successfully!")
    
    def predict(self, new_data):
        """Make predictions on new data"""
        if self.model is None:
            raise ValueError("Model not trained or loaded!")
        
        new_data_scaled = self.scaler.transform(new_data)
        predictions = self.model.predict(new_data_scaled)
        probabilities = self.model.predict_proba(new_data_scaled)
        
        return predictions, probabilities

# Usage Example
if __name__ == "__main__":
    # Initialize pipeline
    pipeline = MLPipeline('dataset.csv')
    
    # Load and preprocess data
    data = pipeline.load_data()
    X, y = pipeline.preprocess(target_column='target')
    
    # Train model
    metrics = pipeline.train_model(X, y)
    
    # Save model
    pipeline.save_model('model.pkl', 'scaler.pkl')

4. JavaScript - Advanced Smooth Scroll Animation

A smooth scroll implementation with easing functions, requestAnimationFrame, and support for both vertical and horizontal scrolling.

/**
 * Advanced Smooth Scroll Animation Library
 * Features: Multiple easing functions, horizontal/vertical scroll, callback support
 */

class SmoothScroll {
  constructor(options = {}) {
    this.duration = options.duration || 1000;
    this.easing = options.easing || 'easeInOutCubic';
    this.offset = options.offset || 0;
    this.onStart = options.onStart || null;
    this.onComplete = options.onComplete || null;
  }

  // Easing functions
  static easings = {
    linear: t => t,
    easeInQuad: t => t * t,
    easeOutQuad: t => t * (2 - t),
    easeInOutQuad: t => t < 0.5 ? 2 * t * t : -1 + (4 - 2 * t) * t,
    easeInCubic: t => t * t * t,
    easeOutCubic: t => (--t) * t * t + 1,
    easeInOutCubic: t => t < 0.5 ? 4 * t * t * t : (t - 1) * (2 * t - 2) * (2 * t - 2) + 1,
    easeInExpo: t => t === 0 ? 0 : Math.pow(2, 10 * (t - 1)),
    easeOutExpo: t => t === 1 ? 1 : -Math.pow(2, -10 * t) + 1,
    easeInOutExpo: t => {
      if (t === 0 || t === 1) return t;
      if (t < 0.5) return Math.pow(2, 20 * t - 10) / 2;
      return (2 - Math.pow(2, -20 * t + 10)) / 2;
    }
  };

  /**
   * Scroll to a specific element
   * @param {string|Element} target - Selector or DOM element
   */
  to(target) {
    const element = typeof target === 'string' 
      ? document.querySelector(target) 
      : target;
    
    if (!element) {
      console.error('Target element not found');
      return;
    }

    const targetPosition = element.getBoundingClientRect().top + window.pageYOffset;
    const startPosition = window.pageYOffset;
    const distance = targetPosition - startPosition - this.offset;
    
    this.animate(startPosition, distance);
  }

  /**
   * Scroll to top of page
   */
  toTop() {
    this.animate(window.pageYOffset, -window.pageYOffset);
  }

  /**
   * Scroll to bottom of page
   */
  toBottom() {
    const distance = document.documentElement.scrollHeight - window.innerHeight - window.pageYOffset;
    this.animate(window.pageYOffset, distance);
  }

  /**
   * Animate scroll
   * @param {number} startPosition - Starting scroll position
   * @param {number} distance - Distance to scroll
   */
  animate(startPosition, distance) {
    const startTime = performance.now();
    const easingFunction = SmoothScroll.easings[this.easing];

    // Call onStart callback
    if (this.onStart) this.onStart();

    const animation = (currentTime) => {
      const elapsed = currentTime - startTime;
      const progress = Math.min(elapsed / this.duration, 1);
      const easedProgress = easingFunction(progress);

      window.scrollTo(0, startPosition + distance * easedProgress);

      if (progress < 1) {
        requestAnimationFrame(animation);
      } else {
        // Call onComplete callback
        if (this.onComplete) this.onComplete();
      }
    };

    requestAnimationFrame(animation);
  }
}

// Usage Examples

// 1. Basic usage
const scroll = new SmoothScroll({ duration: 1200 });
scroll.to('#about-section');

// 2. With callbacks
const scrollWithCallbacks = new SmoothScroll({
  duration: 1500,
  easing: 'easeInOutExpo',
  offset: 80, // Offset for fixed header
  onStart: () => console.log('Scroll started'),
  onComplete: () => console.log('Scroll completed')
});
scrollWithCallbacks.to('#projects');

// 3. Scroll to top
const scrollTop = new SmoothScroll({ duration: 800, easing: 'easeOutCubic' });
scrollTop.toTop();

// 4. Initialize smooth scroll for all anchor links
document.querySelectorAll('a[href^="#"]').forEach(anchor => {
  anchor.addEventListener('click', function(e) {
    e.preventDefault();
    const target = this.getAttribute('href');
    const scroll = new SmoothScroll({ 
      duration: 1000, 
      offset: 70,
      easing: 'easeInOutCubic'
    });
    scroll.to(target);
  });
});

  module.exports = SmoothScroll;
}
